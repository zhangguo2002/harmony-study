# 1.3 ArkUI 基础组件

> ArkUI 是 HarmonyOS 的声明式 UI 框架，所有 UI 都是通过**组合组件**来实现的。

---

## 一、基础显示组件

### Text — 文本

```typescript
Text('Hello HarmonyOS')
  .fontSize(20)           // 字体大小
  .fontColor('#FF0000')   // 字体颜色（支持色值或 Color.Red 等枚举）
  .fontWeight(FontWeight.Bold)  // 字体粗细
  .textAlign(TextAlign.Center)  // 文本对齐
  .maxLines(2)            // 最大行数
  .textOverflow({ overflow: TextOverflow.Ellipsis })  // 超出省略
```

### Image — 图片

```typescript
// 加载本地资源图片（放在 resources/media/ 下）
Image($r('app.media.icon'))
  .width(100)
  .height(100)
  .objectFit(ImageFit.Cover)   // 填充方式
  .borderRadius(8)

// 加载网络图片
Image('https://example.com/photo.jpg')
  .width('100%')
  .aspectRatio(16 / 9)
```

### TextInput — 输入框

```typescript
@State inputValue: string = ''

TextInput({ placeholder: '请输入内容', text: this.inputValue })
  .onChange((value: string) => {
    this.inputValue = value
  })
  .type(InputType.Normal)   // Normal / Password / Email / Number
  .width('80%')
  .height(44)
  .borderRadius(8)
```

### Button — 按钮

```typescript
Button('确认')
  .onClick(() => {
    console.log('按钮点击了')
  })
  .type(ButtonType.Normal)    // Normal / Capsule / Circle
  .width(120)
  .height(44)
  .backgroundColor('#007DFF')
  .fontColor(Color.White)
```

## 二、布局组件

> 布局组件决定子组件如何**排列**。

### Column — 垂直排列

```typescript
Column({ space: 12 }) {   // space: 子组件间距
  Text('第一行')
  Text('第二行')
  Text('第三行')
}
.width('100%')
.padding(16)
.alignItems(HorizontalAlign.Start)   // 子组件水平对齐方式
```

### Row — 水平排列

```typescript
Row({ space: 8 }) {
  Image($r('app.media.icon')).width(40).height(40)
  Text('用户名').fontSize(16)
  Blank()   // 弹性空白，把后面的内容推到右边
  Text('→').fontColor('#999')
}
.width('100%')
.height(56)
.padding({ left: 16, right: 16 })
```

### Stack — 层叠布局

```typescript
Stack({ alignContent: Alignment.BottomEnd }) {
  Image($r('app.media.banner')).width('100%').height(200)
  Text('图片说明')
    .fontColor(Color.White)
    .backgroundColor('rgba(0,0,0,0.4)')
    .padding(8)
}
```

### List — 滚动列表（重要！）

```typescript
@State items: string[] = ['苹果', '香蕉', '橙子', '葡萄', '西瓜']

List({ space: 8 }) {
  ForEach(this.items, (item: string) => {
    ListItem() {
      Row() {
        Text(item).fontSize(16)
      }
      .width('100%')
      .height(50)
      .padding({ left: 16 })
    }
  })
}
.width('100%')
.height('100%')
.divider({ strokeWidth: 1, color: '#EEEEEE' })  // 分割线
```

### Grid — 网格布局

```typescript
Grid() {
  ForEach([1, 2, 3, 4, 5, 6], (item: number) => {
    GridItem() {
      Text(`${item}`)
        .width('100%')
        .aspectRatio(1)
        .backgroundColor('#F0F0F0')
        .textAlign(TextAlign.Center)
    }
  })
}
.columnsTemplate('1fr 1fr 1fr')  // 3列等宽
.rowsGap(8)
.columnsGap(8)
```

## 三、常用修饰属性速查

| 属性 | 说明 | 示例 |
|------|------|------|
| `.width()` | 宽度 | `.width(100)` / `.width('100%')` |
| `.height()` | 高度 | `.height(50)` |
| `.padding()` | 内边距 | `.padding(16)` / `.padding({ top: 8, bottom: 8 })` |
| `.margin()` | 外边距 | `.margin({ top: 12 })` |
| `.backgroundColor()` | 背景色 | `.backgroundColor('#F5F5F5')` |
| `.borderRadius()` | 圆角 | `.borderRadius(8)` |
| `.opacity()` | 透明度 | `.opacity(0.5)` |
| `.visibility()` | 可见性 | `.visibility(Visibility.Hidden)` |
| `.onClick()` | 点击事件 | `.onClick(() => { ... })` |

## 四、综合练习：个人卡片页面

```typescript
@Entry
@Component
struct ProfileCard {
  @State name: string = '张三'
  @State bio: string = 'HarmonyOS 开发者'

  build() {
    Column({ space: 16 }) {
      Image($r('app.media.icon'))
        .width(80)
        .height(80)
        .borderRadius(40)

      Text(this.name)
        .fontSize(22)
        .fontWeight(FontWeight.Bold)

      Text(this.bio)
        .fontSize(14)
        .fontColor('#888888')

      Row({ space: 24 }) {
        Button('关注').width(100).height(36)
        Button('发消息')
          .width(100)
          .height(36)
          .backgroundColor(Color.White)
          .fontColor('#007DFF')
          .border({ width: 1, color: '#007DFF' })
      }
    }
    .width('100%')
    .height('100%')
    .justifyContent(FlexAlign.Center)
    .alignItems(HorizontalAlign.Center)
  }
}
```
