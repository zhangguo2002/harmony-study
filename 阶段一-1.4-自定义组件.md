# 1.4 自定义组件

> 组件化是 ArkUI 开发的核心思想：将可复用的 UI 块封装为独立组件，减少重复代码。

---

## 一、定义自定义组件

使用 `@Component` + `struct` 定义组件：

```typescript
@Component
struct MyCard {
  // 组件内部状态
  @State isLiked: boolean = false

  build() {
    Row() {
      Text('这是一张卡片').fontSize(16)
      Blank()
      Image(this.isLiked ? $r('app.media.liked') : $r('app.media.like'))
        .width(24).height(24)
        .onClick(() => {
          this.isLiked = !this.isLiked
        })
    }
    .width('100%')
    .height(56)
    .padding({ left: 16, right: 16 })
    .backgroundColor(Color.White)
    .borderRadius(8)
  }
}
```

---

## 二、组件间传参：`@Prop`（父→子，单向）

父组件向子组件传递数据，子组件修改不影响父组件：

```typescript
// 子组件
@Component
struct TitleBar {
  @Prop title: string = ''    // 接收父组件传入的值

  build() {
    Text(this.title)
      .fontSize(20)
      .fontWeight(FontWeight.Bold)
  }
}

// 父组件使用
@Entry
@Component
struct HomePage {
  build() {
    Column() {
      TitleBar({ title: '首页' })   // 传入参数
      // ...
    }
  }
}
```

## 三、组件间传参：`@Link`（父↔子，双向）

父子组件共享同一份数据，任一方修改都会同步：

```typescript
// 子组件
@Component
struct SwitchItem {
  @Link isOn: boolean    // 与父组件双向绑定

  build() {
    Toggle({ type: ToggleType.Switch, isOn: this.isOn })
      .onChange((value: boolean) => {
        this.isOn = value   // 子组件修改 → 父组件同步更新
      })
  }
}

// 父组件
@Entry
@Component
struct SettingsPage {
  @State wifiEnabled: boolean = false

  build() {
    Column() {
      SwitchItem({ isOn: $wifiEnabled })  // 注意：传 Link 用 $ 符号
      Text(this.wifiEnabled ? 'WiFi 已开启' : 'WiFi 已关闭')
    }
  }
}
```

> ⚠️ 传递 `@Link` 时用 `$变量名`（加 `$` 前缀）

## 四、`@Builder` 可复用 UI 片段

不需要单独抽成组件，但想复用一段 UI 时，使用 `@Builder`：

```typescript
@Entry
@Component
struct MenuPage {

  @Builder
  MenuItem(icon: Resource, label: string) {
    Row({ space: 12 }) {
      Image(icon).width(24).height(24)
      Text(label).fontSize(16)
      Blank()
      Text('>').fontColor('#CCCCCC')
    }
    .width('100%')
    .height(52)
    .padding({ left: 16, right: 16 })
  }

  build() {
    Column() {
      this.MenuItem($r('app.media.icon'), '个人信息')
      Divider()
      this.MenuItem($r('app.media.icon'), '消息通知')
      Divider()
      this.MenuItem($r('app.media.icon'), '关于我们')
    }
  }
}
```

## 五、完整练习：新闻列表卡片

```typescript
// 数据模型
interface NewsItem {
  title: string
  source: string
  time: string
}

// 子组件：新闻卡片
@Component
struct NewsCard {
  @Prop item: NewsItem = { title: '', source: '', time: '' }

  build() {
    Column({ space: 6 }) {
      Text(this.item.title)
        .fontSize(16)
        .fontWeight(FontWeight.Medium)
        .maxLines(2)
        .textOverflow({ overflow: TextOverflow.Ellipsis })

      Row() {
        Text(this.item.source).fontSize(12).fontColor('#999')
        Blank()
        Text(this.item.time).fontSize(12).fontColor('#999')
      }
      .width('100%')
    }
    .width('100%')
    .padding(16)
    .backgroundColor(Color.White)
    .borderRadius(8)
    .margin({ bottom: 8 })
  }
}

// 页面
@Entry
@Component
struct NewsPage {
  @State newsList: NewsItem[] = [
    { title: 'HarmonyOS 5 正式发布，带来全新开发体验', source: '华为', time: '10分钟前' },
    { title: 'ArkUI 声明式框架深度解析', source: '开发者社区', time: '1小时前' },
    { title: '一次开发多端部署：HarmonyOS 的核心优势', source: '极客时间', time: '2小时前' },
  ]

  build() {
    Column() {
      Text('今日资讯')
        .fontSize(22)
        .fontWeight(FontWeight.Bold)
        .margin({ top: 16, bottom: 12, left: 16 })
        .alignSelf(ItemAlign.Start)

      List() {
        ForEach(this.newsList, (item: NewsItem) => {
          ListItem() {
            NewsCard({ item: item })
          }
        })
      }
      .padding({ left: 16, right: 16 })
    }
    .width('100%')
    .height('100%')
    .backgroundColor('#F5F5F5')
  }
}
```
